public int userCompare(String aName, int aId, String bName, int bId) {
  int result = 0;
  if(aName.compareTo(bName) < 0){
    result = -1;
  }
  else if (aName.compareTo(bName) > 0){
    result = 1;
  }
  else if(aId == bId){
    result = 0;
  }
  else if (aId > bId){
    result = 1;
  }
  else result = -1;
  return result;
}
