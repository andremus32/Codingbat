//AP-1 > copyEndy
public int[] copyEndy(int[] nums, int count) {
  int[] arr = new int[count];
  int arrCount = 0;
  for(int i = 0; i < nums.length; i++){
    if(isEndy(nums[i]) && arrCount < count){
      arr[arrCount] = nums[i];
      arrCount++;
    }
  }
  return arr;
}

public boolean isEndy(int n){
  if((n >= 0 && n <= 10) || (n >= 90 && n <= 100)){
    return true;
  }
  else return false;
}
