//AP-1 > mergeTwo
public String[] mergeTwo(String[] a, String[] b, int n) {
  String[] arr = new String[n];
  int countA = 0;
  int countB = 0;
  for(int i = 0; i < n; i++){
    if(a[countA].compareTo(b[countB]) < 0){
      arr[i] = a[countA];
      countA++;
    }
    else if (a[countA].compareTo(b[countB]) > 0){
      arr[i] = b[countB];
      countB++;
    }
    else {
      arr[i] = a[countA];
      countA++;
      countB++;
    }
  }
  return arr;
}
